package Ecommerce{

    package Utilisateur{

        abstract Personne
        
        class Client
        
        class Vendeur
        
        class Mediateur
        
        class LesComptes
        
        class Compte{
        #int id
        #String nom
        #String prenom
        #String pseudo
        #String email
        #String mdp
        #String adresse
        #int telephone
        #String dateNaissance
        #int numeroCarte
        #String dateExpiration
        #int cryptogramme
        }
    
        package Statistiques{
        
            abstract Stat{
                #int noteMoyennePeriodeActive
                #int noteMoyenneDernierMois
            }
            
            class StatVendeur{
                #int bienProposeeTotal
                #int bienProposeeActuel
            }
            
            class StatClient{
                #int nbAchat
            }
            
            class Note{
                #int valeur
            }
            
            class Commentaire{
                #String com
            }
            
            Stat <|-- StatVendeur
            Stat <|-- StatClient
            Stat -up-"*" Note
            Stat -up-"*" Commentaire
        }
    
        package Litige{
        
            class Litige{
                #String reclamation
            }
            
            class LesLitiges
        
            LesLitiges -right-"*"Litige
        }
        
        LesComptes --"*" Compte
        
        Personne <|-down- Client
        Personne <|-down- Vendeur
        Personne <|-down- Mediateur
        Personne "3"-right-"1" Compte
        
        Mediateur-down-"1"LesLitiges
        Litige-up-"1"Client
        Litige-up-"1"Vendeur
        
        Commentaire --"1" Compte
        Compte --"2" Stat
    }
    
    Package Commerce{
    
        package Produit{
            class Produit{
                #String libelle
                #int reference
		#String description
                #int stockDispo
            }
            
            class LesProduits
            
            package Vente {
                abstract Vente
            
                class VenteNormal{
                    #int prixDeVente
                    #String delaiRetractation
                }
            
                class VenteEnchere{
                    #int prixDepart
                    #int dernierPrixPropose
		    #String dateDebut
                    #String dateFin
                }
                
                Vente <|-- VenteNormal
                Vente <|-- VenteEnchere
            }
            
            LesProduits --"*" Produit
            Produit --"1" Vente
        }
        
        package Commande{
            class LesCommandes
            
            class Commande{
        #int idCommande
        #int idCompteClient	
        #int idCompteVendeur	
        #int prix
        #String delaiExpedition
        #String delaiReclamation
        #String adresseLivraison
        #String delaisNotationCommentaire
            }
            
            LesCommandes --"*" Commande
	    Commande --"*"Produit
        }
    }

    LesCommandes "1"-- Vendeur
    LesProduits "1"-- Vendeur
    Client -- "1" LesCommandes
    StatVendeur -- "1" LesProduits

}
